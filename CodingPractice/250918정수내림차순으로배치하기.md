# 정수 내림차 순으로 배치하기

```Java
class Solution {
    public long solution(long n) {
        
        String s = String.valueOf(n);
        
        // 문자열을 하나씩 자른다. 
        char[] arr = s.toCharArray();

        // 현재 수와 앞자리 수를 비교한다
        for(int i=0; i<arr.length; i++){
            for(int j=i+1;j<arr.length; j++){
                
                // 앞자리수보다 크다면
                if(arr[j]>arr[i]){

                    // 현재 값을 앞으로 위치한다.
                    char temp;
                    temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                    
                }                  
            }
        }
        
        String newStr = new String(arr);
        
        long answer = Long.parseLong(newStr);
        
        return answer;
    }
}
```
