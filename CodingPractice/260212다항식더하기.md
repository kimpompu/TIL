# 다항식 더하기

```Java
class Solution {
    public String solution(String polynomial) {
        int xCount = 0;   
        int numCount = 0; 
        
        String[] terms = polynomial.split(" ");
        
        for (String term : terms) {
            if (term.contains("x")) {
                if (term.equals("x")) {
                    xCount += 1;
                } else {
                    xCount += Integer.parseInt(term.replace("x", ""));
                }
            } else if (!term.equals("+")) {
                numCount += Integer.parseInt(term);
            }
        }
        
        String result = "";
        
        if (xCount > 0) {
            result += (xCount == 1) ? "x" : xCount + "x";
        }
        
        if (numCount > 0) {
            if (result.equals("")) {
                result += numCount; 
            } else {
                result += " + " + numCount; 
            }
        }
        
        return result;
    }
}
```
