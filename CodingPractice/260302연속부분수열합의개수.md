# 연속 부분 수열 합의 개수
```java
import java.util.*;

class Solution {
    public int solution(int[] elements) {
        Set<Integer> set = new HashSet<>();
        int length = elements.length;
        
        for(int start = 0; start < length; start++){
            int currentSum = 0;
            
            for(int len = 1; len <= length; len++){
                currentSum  += elements[(start + len - 1) % length];
                set.add(currentSum);
            }
        }
        
        return set.size();
    }
}
```
