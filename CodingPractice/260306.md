# 요격시스템
```java
import java.util.*;

class Solution {
    public int solution(int[][] targets) {
        
        Arrays.sort(targets, (a, b) -> {
            return a[1] - b[1];
        });
        
        int answer = 0;
        double lastPosition = -1.0;
        
        for(int[] missile : targets){
            int start = missile[0];
            int end = missile[1];
            
            if(start >= lastPosition){
                answer++;
                lastPosition = end;
            }
        }
        
        return answer;
    }
}
```
