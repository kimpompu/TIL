# 제일 작은 수 제거하기

```Java
class Solution {
    public int[] solution(int[] arr) {
        
        // arr 길이가 1일 경우 -1을 가진 배열 반환
        if(arr.length == 1){
            int[] answer = {-1};
            return answer;
        }
        
        // arr보다 길이가 1 작은 배열 선언
        int[] answer = new int [arr.length-1];
        
        int min = arr[0];
        
        for(int i = 1; i<arr.length; i++){
            min = Math.min(min, arr[i]);
        }
        
        int idx = 0;
        
        for(int i=0; i<arr.length; i++){
            if(arr[i]==min){
                //최솟값과 같다면 계속
                continue;
            }
            answer[idx++] = arr[i];
        }

        return answer;
    }
}
```
