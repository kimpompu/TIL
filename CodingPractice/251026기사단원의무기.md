# 기사단원의 무기

```Java
public class Solution {
    public int solution(int number, int limit, int power) {
        int totalIron = 0; // 필요한 철의 총합

        for (int i = 1; i <= number; i++) {
            int divisorCount = countDivisors(i); // 약수 개수 계산

            if (divisorCount > limit) {
                totalIron += power; // 제한 초과 → power로 대체
            } else {
                totalIron += divisorCount; // 그대로 사용
            }
        }

        return totalIron;
    }

    // 약수의 개수를 구하는 함수
    private int countDivisors(int n) {
        int count = 0;

        for (int i = 1; i * i <= n; i++) { // √n까지만 탐색
            if (n % i == 0) {
                count++; // i는 약수
                if (i != n / i) count++; // i와 짝이 되는 n/i도 약수
            }
        }

        return count;
    }
}
```
