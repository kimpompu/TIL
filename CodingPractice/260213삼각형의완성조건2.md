# 삼각형의 완성조건(2)

```Java
import java.util.Arrays;

class Solution {
    public int solution(int[] sides) {
        // 1. 편의를 위해 정렬을 해서 작은 값(a), 큰 값(b)을 구분합니다.
        Arrays.sort(sides);
        int a = sides[0];
        int b = sides[1];
        
        int answer = 0;
        
        // 방법 1: 직접 범위를 순회하며 카운트 (이해하기 쉬운 방식)
        // 위에서 도출한 결론: b - a < x < a + b
        for (int x = b - a + 1; x < a + b; x++) {
            answer++;
        }
        
        return answer;
        
        // Tip: 수학적으로 바로 계산하면 (a + b) - (b - a) - 1 = 2a - 1 입니다.
        // return 2 * a - 1;
    }
}
```
