# 개미 군단
## 그리디 알고리즘
```Java
class Solution {
    public int solution(int hp) {
        
        int answer = 0;
        int general = 0;
        int solider = 0;
        int worker = 0;
        int remains = 0;
        
        general = hp / 5;
        remains = hp % 5;
        
        solider = remains / 3;
        remains = remains % 3;
        
        worker = remains / 1;
        remains = worker % 1;
        
        answer = general + solider + worker;
        
        return answer;
    }
}
```
