# 로또의 최고 순위와 최저 순위

```Java
import java.util.Arrays;

class Solution {
    public int[] solution(int[] lottos, int[] win_nums) {
        int zeroCount = 0;
        int matchCount = 0;

        for (int lotto : lottos) {
            if (lotto == 0) {
                zeroCount++;
            } else if (contains(win_nums, lotto)) {
                matchCount++;
            }
        }

        int best = getRank(matchCount + zeroCount);
        int worst = getRank(matchCount);

        return new int[] {best, worst};
    }

    // 번호가 당첨 번호 배열에 포함되는지 확인
    private boolean contains(int[] arr, int num) {
        for (int a : arr) {
            if (a == num) return true;
        }
        return false;
    }

    // 맞은 개수 → 순위로 변환
    private int getRank(int count) {
        switch (count) {
            case 6: return 1;
            case 5: return 2;
            case 4: return 3;
            case 3: return 4;
            case 2: return 5;
            default: return 6;
        }
    }
}
```
