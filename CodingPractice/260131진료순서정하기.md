# 진료 순서 정하기
```Java
class Solution {
    public int[] solution(int[] emergency) {
        int[] answer = new int[emergency.length];
        
        for(int i=0; i<emergency.length;i++){
            int rank = 1;
            for(int j=0; j<emergency.length;j++){
                if(emergency[i]<emergency[j]){
                    rank++;
                }
            }
            answer[i] = rank;
        }
        
        return answer;
    }
}
```

```Java
import java.util.*;
import java.util.stream.Collectors;

class Solution {
    public int[] solution(int[] emergency) {
        // 리스트 내림차순 정렬
        List<Integer> sorted = Arrays.stream(emergency) // 1. 배열을 스트림으로 변환
            .boxed()           // 2. int를 Integer 객체로 변환
            .sorted(Comparator.reverseOrder()) // 3. 내림차순 정렬
            .collect(Collectors.toList());      // 4. 리스트로 저장

        return Arrays.stream(emergency) // 1. 원본 배열을 다시 스트림으로
      .map(e -> sorted.indexOf(e) + 1) // 2. 값을 인덱스 번호로 변환
      .toArray(); // 3. 다시 배열로 변환
    }
}
```
